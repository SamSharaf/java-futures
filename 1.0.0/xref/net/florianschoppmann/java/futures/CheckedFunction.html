3   import javax.annotation.Nullable;

5   /**
6    * Represents a function that accepts one argument and produces a result. The function may throw a checked exception.
7    *
8    * <p>This is a functional interface whose functional method is {@link #checkedApply(Object)}. It is similar to
9    * {@link java.util.function.Function}, except that the functional method may throw a checked exception.
10   *
11   * @param <T> the type of the input to the function
12   * @param <R> the type of the result of the function
13   *
14   * @see java.util.function.Function
15   */
16  @FunctionalInterface
17  public interface CheckedFunction<T, R> {
18      /**
19       * Applies this function to the given argument.
20       *
21       * @param argument the function argument
22       * @return the function result
23       * @throws Exception if unable to compute a result
24       */
25      @Nullable
26      R checkedApply(@Nullable T argument) throws Exception;
27  }
