5   /**
6    * Represents an operation that accepts two input arguments and returns no result. Unlike most other functional
7    * interfaces, {@code CheckedBiConsumer} is expected to operate via side-effects.
8    *
9    * <p>This is a functional interface whose functional method is {@link #checkedAccept(Object, Object)}. It is similar to
10   * {@link java.util.function.BiConsumer}, except that the functional method may throw a checked exception.
11   *
12   * @param <T> the type of the first argument to the operation
13   * @param <U> the type of the second argument to the operation
14   *
15   * @see java.util.function.BiConsumer
16   */
17  @FunctionalInterface
18  public interface CheckedBiConsumer<T, U> {
19      /**
20       * Performs this operation on the given arguments.
21       *
22       * @param first the first input argument
23       * @param second the second input argument
24       * @throws Exception if unable to process the input
25       */
26      void checkedAccept(@Nullable T first, @Nullable U second) throws Exception;
27  }
