49      private static Throwable extractException(CompletableFuture<?> future) {
50          Assert.assertTrue(future.isDone());
51          AtomicReference<Throwable> failureReference = new AtomicReference<>();
52          future.whenComplete((result, failure) -> {
53              failureReference.set(failure);
54          });
55          @Nullable Throwable throwable = failureReference.get();
56          Assert.assertNotNull(throwable);
57          return throwable;
58      }

92      private static <T> CompletableFuture<T> assertNoShortCircuit(
93              TriFunction<CompletableFuture<Integer>, CompletableFuture<Integer>, CompletableFuture<Integer>,
94                  CompletableFuture<T>> function) {
95          CompletableFuture<Integer> first = new CompletableFuture<>();
96          CompletableFuture<Integer> second = new CompletableFuture<>();
97          CompletableFuture<Integer> third = new CompletableFuture<>();
98          CompletableFuture<T> combinedFuture = function.apply(first, second, third);
99          second.completeExceptionally(new ExpectedException(2));
100         first.completeExceptionally(new ExpectedException(1));
101 
102         Assert.assertFalse(combinedFuture.isDone());
103         third.complete(3);
104         Assert.assertTrue(combinedFuture.isCompletedExceptionally());
105         assertCompletedWithComplEx(combinedFuture, 1);
106 
107         // Ensure that no double-wrapping in a CompletionException happens
108         CompletableFuture<Integer> zeroth
109             = Futures.completedExceptionally(new CompletionException(new ExpectedException(0)));
110         assertCompletedWithComplEx(function.apply(zeroth, first, second), 0);
111 
112         return combinedFuture;
113     }

370     private abstract static class Try<T> {
371         private Try() {
372             assert getClass().equals(Success.class) || getClass().equals(Failure.class);
373         }
374     }

376     private static final class Success<T> extends Try<T> {
377         private final T result;
378 
379         private Success(T result) {
380             this.result = result;
381         }
382 
383         @Override
384         public boolean equals(@Nullable Object otherObject) {
385             return this == otherObject
386                 || (otherObject instanceof Success<?> && result.equals(((Success<?>) otherObject).result));
387         }
388 
389         @Override
390         public int hashCode() {
391             return result.hashCode();
392         }
393     }

396         private final Throwable throwable;

403     private static final class TryContainer {
404         @Nullable private volatile Try<Integer> integerTry;
405 
406         private void consume(@Nullable Integer result, @Nullable Throwable throwable) {
407             assert result != null || throwable != null;
408             integerTry = throwable != null
409                 ? new Failure<>(throwable)
410                 : new Success<>(result);
411         }
412 
413         private void consumeAndThrow(@Nullable Integer result, @Nullable Throwable throwable) {
414             consume(result, throwable);
415             throw new ExpectedException(WHEN_COMPLETE_FAILED);
416         }
417     }

583     private static void assertTryWithResourceResult(CompletableFuture<Integer> future, AutoClosableImpl autoClosable,
584             InitialStageKind initialStageKind, @Nullable int[] expectedExceptionIds) {
585         if (expectedExceptionIds == null) {
586             Assert.assertTrue(future.isDone());
587             Assert.assertEquals((int) future.join(), DEFAULT_RESULT);
588         } else {
589             assert expectedExceptionIds.length > 0;
590             int[] suppressedIds = Arrays.copyOfRange(expectedExceptionIds, 1, expectedExceptionIds.length);
591             assertCompletedWithComplEx(future, expectedExceptionIds[0], suppressedIds);
592         }
593         if (initialStageKind == InitialStageKind.SUCCESS) {
594             Assert.assertFalse(autoClosable.open);
595         }
596     }

661     private static CompletableFuture<Integer> newComposeWithResourceFuture(
662             Behavior futureBehavior, WrapKind wrapKind) {
663         if (futureBehavior == Behavior.EXCEPTION) {
664             Throwable throwable = wrapKind == WrapKind.WRAP
665                 ? new CompletionException(new ExpectedException(NEW_FUTURE_FAILED))
666                 : new ExpectedException(NEW_FUTURE_FAILED);
667             return Futures.completedExceptionally(throwable);
668         } else {
669             return CompletableFuture.completedFuture(DEFAULT_RESULT);
670         }
671     }
